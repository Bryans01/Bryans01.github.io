def write_to_index(path_to_blog, path_to_new_content): with open(path_to_blog/"index.html") as index: soup = Soup(index, features="lxml") links = soup.find_all("a") last_link = links[-1] if check_for_duplicate_links(path_to_new_content, links): raise ValueError("Link does already exist!") link_to_new_blog = soup.new_tag("a", href=Path(*path_to_new_content.parts[-2:])) link_to_new_blog.string = path_to_new_content.name.split(".")[0] last_link.insert_after(link_to_new_blog) with open(path_to_blog/"index.html", "w") as f: f.write(str(soup.prettify(formatter='html')))
